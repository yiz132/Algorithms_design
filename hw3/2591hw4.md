```

job = [
        [1, 2, 40],
        [2, 4, 15],
        [3, 3, 60],
        [4, 2, 20],
        [5, 3, 10],
        [6, 1, 45],
        [7, 1, 55]
    ]

n = len(job)

for i in range(1, n):
    for j in range(0, n-i):
        if job[j+1][2]>job[j][2]:
            temp = job[j+1]
            job[j+1] = job[j]
            job[j] = temp
# print(job)

dmax = 0
for j in range(n):
    if job[j][1] > dmax:
        dmax = job[j][1]
# print(dmax)

timeslot = [-1] * dmax
filledTimeSlot = 0

for k in range(n+1):
    m = min(dmax, job[k][1])
    while(m >=1):
        if timeslot[m-1] == -1: 
            timeslot[m-1] = k
            filledTimeSlot = filledTimeSlot + 1
            break
        m = m - 1
    
    if filledTimeSlot == dmax:
        break

# print(timeslot)

profit = 0

for l in range(dmax):
    id = timeslot[l]
    profit = profit + job[id][2]

# print(profit)
```
`total profit: 170`

```
def maxProfix(input):
    input = sorted(input,key = lambda s:s[1], reverse = True)
    days = set()
    res = 0
    for arr in input:
        if days.__contains__(arr[0]):
            continue
        res += arr[1]
        days.add(arr[0])
    return res

input = [[2,40], [4,15], [3,60], [2,20], [3,10], [1,45],[1,55]]
print(maxProfix(input))
```
